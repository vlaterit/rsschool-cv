function mygcd(x, y) {
	let dividend = x > y ? x : y;
	let divisor = x < y ? x : y;

	while(dividend % divisor !== 0) {
		const reminder = dividend % divisor;
		dividend = divisor;
		divisor = reminder;
	}

	return divisor;
}
